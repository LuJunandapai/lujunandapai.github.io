# Apache ECharts 图表

说明: 一个基于 JavaScript 的开源可视化图表库 快速入门open in new window所有示例open in new window

# 安装 ECharts

Apache ECharts: CSDN 参考地址open in new window

#（1）安装echarts包
npm install echarts --save
cnpm install echarts --save
#（2）安装vue echarts工具包
npm install echarts vue-echarts
cnpm install echarts vue-echarts

# Mian.js 引入

import { createApp } from 'vue'
import App from './App.vue'
import router from './router'
// Element Plus 引入
import ElementPlus from 'element-plus';
import "element-plus/dist/index.css";
import * as ElementPlusIconsVue from "@element-plus/icons-vue";
// 引入 axios
import axios from 'axios';

import ECharts from 'vue-echarts'  // 引入ECharts
import "echarts";                  // 全局引入echarts

const app = createApp(App);
for (const [key, component] of Object.entries(ElementPlusIconsVue)) {
    app.component(key, component);
}

app.use(router);
app.use(ElementPlus);
app.provide('$axios',axios);
// 挂载ECharts 参数1:vue文件中挂载echarts的名字  参数2:挂载的组件
app.component('ECharts',ECharts)
app.mount("#app");

# ECharts 使用

**情形一：**一个页面中只使用1~2个ECharts图表

<template>
  <div id="MyTest">
    <!-- 图表的div标签 ref:方法数据指定名称 id:样式 -->
    <div ref="apaichart" id="demoDIV"></div>
  </div>
</template>

<script>
// 引入echarts的位置
import * as echarts from "echarts";

export default {
  name: 'MyTest',
  data: function () {
    return {
      // 图表的数据 会根据官网提供数据格式显示对应的图表
      option: {
        title: {
          text: 'World Population'
        },
        tooltip: {
          trigger: 'axis',
          axisPointer: {
            type: 'shadow'
          }
        },
        legend: {},
        grid: {
          left: '3%',
          right: '4%',
          bottom: '3%',
          containLabel: true
        },
        xAxis: {
          type: 'value',
          boundaryGap: [0, 0.01]
        },
        yAxis: {
          type: 'category',
          data: ['Brazil', 'Indonesia', 'USA', 'India', 'China', 'World']
        },
        series: [
          {
            name: '2011',
            type: 'bar',
            data: [18203, 23489, 29034, 104970, 131744, 630230]
          },
          {
            name: '2012',
            type: 'bar',
            data: [19325, 23438, 31000, 121594, 134141, 681807]
          }
        ]
      },
    };
  },
  mounted() {
    // 页面加载会执行此方法进行图表数据赋值初始化显示
    this.getDate();
  },
  methods: {
    getDate() {
      // 指定图表HTML标签ref值
      let myEcharts = echarts.init(this.$refs.apaichart);
      // 设置图表参数
      myEcharts.setOption(this.option);
    }
  },
}
</script>

情形二：一个页面中使用了多个ECharts图表

  • 封装图表组件
  • 使用封装的组件名称进行引入渲染展示

注意：如果要在一个页面中使用多个ECharts图表，就会出现很多很多option和类名chart，容易混乱，所以最好把图表们封装成组件使用

  <div id="HomePageView">

  name: 'HomePageView',
  components: {MyTest, ZhuZuangView, ChartView},
  data: function () {

# 子组件

  • EChartsView.vue 图表组件模板
<template>
  <div id="EChartsView">
    <!-- 图表的div标签 ref:方法数据指定名称 id:样式 -->
    <div ref="chart" id="chartDiv"></div>
  </div>
</template>

<script>
// 引入echarts的位置
import * as echarts from "echarts";

export default {
  name: 'EChartsView',
  data: function () {
    return {
      // 图表的数据 会根据官网提供数据格式显示对应的图表
      chartData: {},
    };
  },

  // 页面加载会执行此方法进行图表数据赋值初始化显示
  mounted() {
    // this.getDate();
  },

  methods: {
    // 通过获取data值渲染
    setCharts() {
      // 指定图表HTML标签ref值
      let myEcharts = echarts.init(this.$refs.chart);
      // 设置图表参数
      myEcharts.setOption(this.chartData);
    },
    // 直接传递值进行渲染
    setChartsData(chartData) {
      // 指定图表HTML标签ref值
      let myEcharts = echarts.init(this.$refs.chart);
      // 设置图表参数
      myEcharts.setOption(chartData);
    }
  },
}
</script>

# ECharts 后端案例

  • 后端调用获取数据, 在请求成功获取数据后再调用子组件
  • 子组件为 ECharts 数据传递子组件渲染模板
<template>
  <div id="tubiao">
    <el-button type="primary" @click="getBingViewMethods()">刷新按钮</el-button>
    <EChartsView ref="bing" style="width: 40%; height: 300px;"/>
  </div>
</template>

<script>
import {getBingTestData} from "@/http/testApi";
import EChartsView from "@/views/ApacheECharts/EChartsView";

export default {
  name: "tubiao",
  components: {EChartsView},
  // data: 对象就是要渲染到页面上的数据 | 用于cmd创建的vue
  data: function () {
    return {
      socketData: '',
      // 图表数据 可传递到子图表组件当中进行渲染
      bingData: {
        title: {
          text: '饼状图表-测试',
          subtext: '调用后端接口数据提供',
          left: 'center'
        },
        tooltip: {
          trigger: 'item'
        },
        legend: {
          orient: 'vertical',
          left: 'left'
        },
        series: [
          {
            name: 'Access From',
            type: 'pie',
            radius: '50%',
            data: [
                //图表数据
            ],
            emphasis: {
              itemStyle: {
                shadowBlur: 10,
                shadowOffsetX: 0,
                shadowColor: 'rgba(0, 0, 0, 0.5)'
              }
            }
          },
        ]
      },
    }
  },

  // moubted: 页面加载完毕后执行
  mounted() {
    this.getBingViewMethods();
  },

  // 组件销毁之前触发
  beforeDestroy() {

  },

  // 在离开当前路由前执行清理工作 例如关闭 WebSocket 连接等
  beforeRouteLeave(to, from, next) {

  },

  // methods: 对象中的方法，可以在页面中调用
  methods: {
    // 刷新图表数据
    getBingViewMethods() {
      // 调用API then_正常执行 / catch_执行异常
      getBingTestData().then(res => {
        console.log(res.data);
        this.bingData.series[0].data = res.data
        this.$refs.bing.setChartsData(this.bingData);
      }).catch((erro) => {
        console.log(erro, 'erro')
      });
    },
  },
}
</script>

# WebSocket 实时图表

  • 后端定时器轮询获取数据使用 WebSocket 发送数据 (后端放行)
  • 前端获取到数据实时调用子组件赋值渲染更新图表
  • WebSocket 进行赋值使用 this.$nextTick 否则可能报错
  • 子组件为 ECharts 数据传递子组件渲染模板
<template>
  <div id="tubiao">
    <!--<h1>服务器发送的数据: {{ socketData }}</h1>-->
    <el-button type="primary" @click="getBingViewMethods()">刷新按钮</el-button>
    <EChartsView ref="bing" style="width: 40%; height: 300px;"/>
  </div>
</template>

<script>
import {getBingTestData} from "@/http/testApi";
import EChartsView from "@/views/ApacheECharts/EChartsView";

export default {
  name: "tubiao",
  components: {EChartsView},
  // data: 对象就是要渲染到页面上的数据 | 用于cmd创建的vue
  data: function () {
    return {
      socketData: '',
      // 图表数据 可传递到子图表组件当中进行渲染
      bingData: {
        title: {
          text: '饼状图表-测试',
          subtext: '调用后端接口数据提供',
          left: 'center'
        },
        tooltip: {
          trigger: 'item'
        },
        legend: {
          orient: 'vertical',
          left: 'left'
        },
        series: [
          {
            name: 'Access From',
            type: 'pie',
            radius: '50%',
            data: [],
            emphasis: {
              itemStyle: {
                shadowBlur: 10,
                shadowOffsetX: 0,
                shadowColor: 'rgba(0, 0, 0, 0.5)'
              }
            }
          },
        ]
      },
    }
  },

  // moubted: 页面加载完毕后执行
  mounted() {
    this.getBingViewMethods();
    // 链接 WebSocket
    if ('WebSocket' in window) {
      let userInfo = JSON.parse(window.localStorage.getItem('account'));
      this.websocket = new WebSocket('ws://localhost:9000/webSocket/' + userInfo.id)
      this.initWebSocket()
    } else {
      alert('当前浏览器 Not support websocket')
    }
  },

  // 组件销毁之前触发
  beforeDestroy() {
    console.log("关闭页面")
    this.onbeforeunload()
  },

  // 在离开当前路由前执行清理工作 例如关闭 WebSocket 连接等
  beforeRouteLeave(to, from, next) {
    console.log("离开当前路由");
    this.closeWebSocket();
    next(); // 表示允许导航离开，即不进行阻止
  },

  // methods: 对象中的方法，可以在页面中调用
  methods: {
    // webSocket 系列方法
    initWebSocket() {
      // 连接错误
      this.websocket.onerror = this.setErrorMessage
      // 连接成功
      this.websocket.onopen = this.setOnopenMessage
      // 收到消息的回调
      this.websocket.onmessage = this.setOnmessageMessage
      // 连接关闭的回调
      this.websocket.onclose = this.setOncloseMessage
      // 监听窗口关闭事件，当窗口关闭时，主动去关闭websocket连接，防止连接还没断开就关闭窗口，server端会抛异常。
      window.onbeforeunload = this.onbeforeunload
    },
    setErrorMessage() {
      console.log('WebSocket连接发生错误 状态码：' + this.websocket.readyState)
    },
    setOnopenMessage() {
      console.log('WebSocket连接成功 状态码：' + this.websocket.readyState)
    },
    setOnmessageMessage(event) {
      // 根据服务器推送的消息做自己的业务处理
      this.socketData = event.data;
      console.log('服务端返回：' + event.data);

      let resData = JSON.parse(event.data);
      // 确保 DOM 更新后再使用 this.$refs.bing 否则可能报错
      this.$nextTick(() => {
        this.bingData.series[0].data = resData;
        this.$refs.bing.setChartsData(this.bingData);
      });
    },
    setOncloseMessage() {
      console.log('WebSocket连接关闭 状态码：' + this.websocket.readyState)
    },
    onbeforeunload() {
      this.closeWebSocket()
    },
    closeWebSocket() {
      this.websocket.close()
    },

    // 刷新图表数据
    getBingViewMethods() {
      // 调用API then_正常执行 / catch_执行异常
      getBingTestData().then(res => {
        console.log(res.data);
        this.bingData.series[0].data = res.data
        this.$refs.bing.setChartsData(this.bingData);
      }).catch((erro) => {
        console.log(erro, 'erro')
      });
    },
  },
}
</script>
