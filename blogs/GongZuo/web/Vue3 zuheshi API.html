在 vite.config.js 配置

import { fileURLToPath, URL } from 'node:url'

import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'

// 1. Element Plus 按需引入
import AutoImport from 'unplugin-auto-import/vite'
import Components from 'unplugin-vue-components/vite'
import { ElementPlusResolver } from 'unplugin-vue-components/resolvers'

// 官网: https://vitejs.dev/config/
export default defineConfig({
  plugins: [
    vue(),
    //  2. Element Plus 按需引入配置 AutoImport / Components
    AutoImport({
      resolvers: [ElementPlusResolver()],
    }),
    Components({
      resolvers: [ElementPlusResolver()],
    }),
  ],
  // 相当于 index.js 中的 import.meta.env.BASE_URL 配置所有路由的前缀 默认为 /
  base: '/',
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url))
    }
  }
})

# 表单验证

<script setup>
import { ref } from 'vue'

// 表单的ref对象
const userFormRef = ref(null)
// 表单数据对象
const userForm = ref({
  username: '',
  password: ''
})
// 表单验证方法 自定义函数写在校验对象上面
// 参数: rule 规则 | value 表单值 | callback 回调函数 无论校验成功与否都需要callback
const validateUsername = (rule, value, callback) => {
    if (value === '') {
      // 校验失败 返回错误信息
      callback(new Error('请输入账户'))
    } else {
      // 校验成功 无返回值
      callback()
    }
}
// 表单验证规则 { 规则规则, 提示信息, 触发事件 }
// 1. min: 3, max: 5, 长度校验
// 2. required: true, 必填项不能为空
// 3. pattern: /^[a-zA-Z0-9_-]{4,16}$/, 正则校验
// 4. validator: 自定义校验规则
// message: '请输入账户' 为提示信息
// trigger: '触发事件' 类型: blur_失去焦点, change_改变值, input_输入值 三种
const rules = ref({
  username: [
    // { required: true, message: '请输入账户', trigger: 'blur' },
    { min: 3, max: 5, message: '长度在 3 到 5 个字符', trigger: 'blur' },
    { pattern: /^[a-zA-Z0-9_-]{4,16}$/, message: '请输入正确的账户', trigger: 'blur' },
    // 自定义校验规则 以外部函数的形式定义
    { validator: validateUsername, trigger: 'blur' }
  ],
  password: [
    { min: 3, max: 5, message: '长度在 3 到 5 个字符', trigger: 'blur' },
    // 自定义校验规则 以内部函数的形式定义
    {
      validator: (rule, value, callback) => {
        if (value === '') {
          callback(new Error('请输入密码'))
        } else {
          callback()
        }
      },
      trigger: 'blur'
    }
  ]
})
// 表单提交按钮
const btnlogin = () => {

}
// 表单重置按钮
const resetForm = () => {

}

</script>

<template>
  <main>
    <el-row type="flex" class="row-bg" justify="center" align="middle">
      <el-col :span="5">
        <!-- 1. 表单 :rules="rules" 绑定校验对象 -->
        <el-form :model="userForm" status-icon :rules="rules" ref="userFormRef">
          <!-- 2. 输入框 prop="username" 指定输入框字段 -->
          <el-form-item label="账户" prop="username">
            <el-input type="text" v-model="userForm.username"></el-input>
          </el-form-item>
          <el-form-item label="密码" prop="password">

          </el-form-item>
        </el-form>
      </el-col>
    </el-row>
  </main>

# 基础语法

  • 声明数据 state / 声明方法 action / 声明计算属性 getter / 暴露数据和方法 return 否则无法使用
  • 导入js后就可获取对象 直接操作数据和方法
import { defineStore} from "pinia";
import {computed, ref} from "vue";

// 定义 store
// useCounterStore -> use名称Store 规范
// defineStore(仓库的唯一标识, ()  => { })
export const useCounterStore = defineStore('counter', () => {
    // 声明数据 state
    const count = ref(99)
    const name = ref('张三')
    // 声明方法 action
    const increment = () => {
        count.value++
    }
    const decrement = () => {
        count.value--
    }
    // 声明计算属性 getter
    const doubleCount = computed(() => count.value * 2)
    // 暴露数据和方法 return 否则无法使用
    return {
        count,
        name,
        increment,
        decrement,
        doubleCount
    }
})

import {defineStore} from "pinia";
import {ref} from "vue";

// 定义 store
// defineStore(仓库的唯一标识, ()  => { })
export const usegetDataStore = defineStore('getData', () => {
    // 声明数据 state
    const arr = ref([1, 2, 3])
    // 异步声明方法进行操作数据 action
    const getdata = async () => {
        console.log('执行');
        const adta = await axios.get('http://127.0.0.1:9001/cnareaChina/getCnareaChinaById');
        console.log(adta, 'adta');
        arr.value = adta;
    }
    // 声明计算属性 getter

    // 暴露数据和方法 return 否则无法使用
    return {
        arr,
        getdata
    }
})

# storeToRefs 响应式

  • 直接 解构 获取store中的数据: 会失去响应式/双向绑定
  • 使用 storeToRefs() 方法, 将store中的数据转换为响应式数据
<!-- setup: 允许使用vue3的组合式API-->
<script setup>
import { ref } from 'vue'
// 导入store
import { useCounterStore } from '@/store/counter.js'
import {storeToRefs} from "pinia";
// 获取store对象
const counterStore = useCounterStore();

// 1. 直接 解构 获取store中的数据: 会失去响应式/双向绑定
// const { count, doubleCount } = counterStore;

// 2. 使用 storeToRefs() 方法, 将store中的数据转换为响应式数据
const { count, doubleCount } = storeToRefs(counterStore);

</script>

<template>
  <div>
    <h1>store 数据 - {{counterStore.count}} - {{counterStore.doubleCount}}</h1>
    <!--<h1>解构 store 数据 - {{count}} - {{doubleCount}}</h1>-->
    <h1>storeToRefs store 数据 - {{count}} - {{doubleCount}}</h1>
    <button @click="counterStore.increment">+++</button>
    <button @click="counterStore.decrement">---</button>
  </div>
</template>

import { defineStore} from "pinia";

# 响应式 Reactive - Ref

reactive

  • reactve: 接收对象数据, 返回响应式数据 (数据变化, 视图自动更新)

ref

  • ref: 接收简单类型或者对象类型的数据传入并返回一个响应式的对象 (数据变化, 视图自动更新)
  • 本质：是在原有传入数据的基础上，外层包了一层对象，包成了复合类型，然后再借助reactive实现的响应式
  • 底层，包成复杂类型之后，再借助reactive实现的响应式
  • 在 script setup 中，通过 名称.value 获取 ref 对象中的值
  • 在 template 中，直接使用 名称 获取 ref 对象中的值
<!-- setup: 允许使用vue3的组合式API-->

# 计算属性 Computed

用来处理数据的方法, 计算属性基本思想和Vue2的完全一致，组合式API下的计算属性只是修改了写法

核心步骤：

  • 导入computed函数
  • 执行函数在回调参数中return基于响应式数据做计算的值，用变量接收

注意点:

  • 计算属性中不应该有“副作用”, 比如异步请求/修改dom
  • 避免直接修改计算属性的值, 计算属性应该是只读的，特殊情况可以配置get set
<!-- setup: 允许使用vue3的组合式API-->
<script setup>
// computed: 计算属性
import {computed, ref} from 'vue';
// 声明响应式数据 数组
const arr = ref([1, 2, 3, 7, 10, 8, 35, 41, 22, 64]);
// 计算属性
const evenArr = computed(() => {
  // 过滤出偶数
  return arr.value.filter(item => item % 2 === 0);
});
// 定义函数, 用于添加数据
const addData = () => {
  arr.value.push(Math.floor(Math.random() * 100));
};
</script>

<template>
  <div>
    <h2>计算属性</h2>
    <div>原始数据: {{ arr }}</div>
    <div>偶数数据: {{ evenArr }}</div>
    <button @click="addData">添加数据</button>
  </div>
</template>

# 侦听器函数 Watch

  • 作用：侦听一个或者多个数据的变化，数据变化时执行回调函数
  • 俩个额外参数：1.immediate（立即执行）2.deep（深度侦听）
  • 侦听器: immediate: 进入页面是否立即执行 / deep: 是否深度侦听(默认是浅层侦听)
  • ref(简单数据类型) 可直接侦听, ref(复杂数据类型) 侦听整个对象内存地址,侦听不到内部数据的变化
  • deep: true 可以侦听ref(复杂数据类型)到内部数据的变化
  • 精确侦听对象的某个属性
<!-- setup: 允许使用vue3的组合式API-->
<script setup>

import {ref, watch} from "vue";

const count = ref(0);
const increment = () => {
  count.value++;
};
const name = ref('Vue3');
const changeName = () => {
  name.value = 'Vue3' + Math.floor(Math.random() * 10);
};
const user = ref({
  name: 'Vue3',
  age: 18
});

// 侦听器: 单个数据的变化 newValue: 新值, oldValue: 旧值
watch(count, (newValue, oldValue) => {
  console.log('count: ', newValue, oldValue);
});

// 侦听器: 多个数据的变化 [newValue1, newValue2]: 新值, [oldValue1, oldValue2]: 旧值
watch([count, name], ([countValue, nameValue], [countOldValue, nameOldValue]) => {
  console.log('count: ', countValue, countOldValue);
  console.log('name: ', nameValue, nameOldValue);
});

// 侦听器: immediate: 进入页面是否立即执行 deep: 是否深度侦听(默认是浅层侦听)
// ref(简单数据类型) 可直接侦听, ref(复杂数据类型) 侦听整个对象内存地址,侦听不到内部数据的变化
// deep: true 可以侦听ref(复杂数据类型)到内部数据的变化
watch(user, (newValue, oldValue) => {
  console.log('count: ', newValue, oldValue);
}, {immediate: true, deep: true});

// 精确侦听对象的某个属性
// 需求：在不开后deep的前提下，侦听age的变化，只有age变化时才执行回调
watch(() => user.value.age, (newValue, oldValue) => {
  console.log('count: ', newValue, oldValue);
});
</script>

<template>
  <div>
    <div>{{ count }}</div>
    <button @click="increment">点击增加</button>
    <div>{{ name }}</div>
    <button @click="changeName">点击修改</button>
  </div>
</template>

// 1. 跨层传递普通数据 provide('变量名称', '数据')
provide('name', '张三')
// 2. 跨层传递响应式数据 注意: 双向绑定同步更新但是子组件不能更改父组件的数据
const age = ref(18)
provide('age', age)
// 3. 跨层传递方法 注意: 可以通过方法传递数据更改父组件的数据
provide('changeAge', (newAge) => {
  age.value = newAge;
})

</script>

<template>
  <div>
    <h1>一级组件 层层嵌套 跨组件数据传递</h1>
    <tow></tow>
  </div>
</template>

<!-- setup: 允许使用vue3的组合式API-->
<script setup>

// 使用 inject('变量名称') 获取跨层传递的数据
import {inject} from "vue";

const name = inject('name')
const age = inject('age')
const changeAge = inject('changeAge')
const up = () => {
  changeAge(20)
}

</script>

<template>
  <div>
    <h1>三级组件</h1>
    <button @click="up">更改年龄</button>
    <button @click="changeAge(90)">更改年龄</button>
    <p>一级的数据: {{name}} - {{age}}</p>
  </div>
</template>

# 路由 API

<script setup>
import { RouterView } from 'vue-router'

import { useRoute, useRouter } from "vue-router";
// 在仕 Vue3 CompositionApI 中
// 1.获取路由对象 router useRouter
// 2.获取路由参数 route useRoute
const router = useRouter();
const route = useRoute();
const goto = () => {
  // 调用方法直接使用 this.$router.push('/about') 无法跳转
  // router.push('/about') 跳转路由
  router.push('/about')
}
</script>

<template>
  <div>
    <router-link to="/about">跳转标签</router-link>
    <button @click="$router.push('/about')">跳转按钮</button>
    <button @click="goto">跳转按钮 调用方法this无法跳转</button>
  </div>
  <!-- APP 路由组件 -->
  <RouterView />
</template>
